## docker-compose.yml to run DB instance

    version: '3.8'

    services:
    db:
        image: mcr.microsoft.com/mssql/server:2022-latest
        container_name: mssql_server
        restart: always
        environment:
        ACCEPT_EULA: "Y"
        MSSQL_SA_PASSWORD: "AppUser@123" # Use a strong password
        MSSQL_PID: "Express"
        ports:
        - "1433:1433"
        volumes:
        - mssql_data:/var/opt/mssql
        healthcheck:
        test: ["CMD", "/opt/mssql-tools18/bin/sqlcmd", "-S", "localhost", "-U", "sa", "-P", "AppUser@123", "-Q", "SELECT 1", "-No"]
        interval: 10s
        timeout: 5s
        retries: 10

    db-init:
        image: mcr.microsoft.com/mssql/server:2022-latest
        container_name: mssql_init
        restart: "no"
        depends_on:
        db:
            condition: service_healthy
        volumes:
        - ./init.sql:/init.sql:ro
        # This command runs the script and then the container exits
        command: /opt/mssql-tools18/bin/sqlcmd -S db -U sa -P "AppUser@123" -i /init.sql -C -No

    volumes:
    mssql_data: # This ensures your data persists on the server's disk


  ## command to run DB instance 
    docker-compose up -d


  ## updated .env for backend
    PORT=5000
    DB_USER=sa
    DB_PASSWORD=AppUser@123             # Must match MSSQL_SA_PASSWORD on Server B
    DB_SERVER=192.168.1.50             # <--- Change this to Server B's IP address
    DB_DATABASE=FormDb1                # Must match the DB created in init.sql
    JWT_SECRET=super_secret_jwt_key_2026
